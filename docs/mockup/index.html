Quick Start

# 1. Clone & Install
git clone https://github.com/rofilenewseong-ui/petholo.git
cd petholo
pnpm install

# 2. Environment Setup
cp .env.example .env
# Edit .env with your Firebase, JWT, AI API keys

# 3. Development
pnpm dev          # Start all (web:3000 + api:4000)
pnpm dev:web      # Frontend only
pnpm dev:api      # Backend only

# 4. Build
pnpm build        # Build all apps

# 5. View Mockups
open docs/mockup/index.html
